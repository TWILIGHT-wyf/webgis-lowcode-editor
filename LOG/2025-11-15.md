### Component接口：
```ts
export interface component {
  id: string
  type: string
  position: { x: number; y: number }
  size: { width: number; height: number }
  rotation: number
  zindex: number
  // 通用扩展属性容器（组件特定属性，如文本内容、颜色等）
  props?: Record<string, any>

  // 数据来源（可选，用于动态数据绑定）
  dataSource?: {
    type: 'static' | 'api' | 'database' | 'websocket'  // 数据类型
    url?: string  // API URL
    method?: 'GET' | 'POST' | 'PUT' | 'DELETE'  // HTTP 方法
    headers?: Record<string, string>  // 请求头
    params?: Record<string, any>  // 查询参数
    body?: any  // 请求体
    refreshInterval?: number  // 自动刷新间隔（毫秒）
    transform?: string  // 数据转换脚本（可选）
  }

  // 组件联动（可选，定义与其他组件的交互）
  linkages?: {
    trigger: {
      event: string  // 触发事件，如 'click', 'change'
      conditions?: Record<string, any>  // 触发条件
    }
    actions: {
      targetId: string  // 目标组件 ID
      action: 'updateProps' | 'show' | 'hide' | 'animate' | 'setData'  // 动作类型
      params?: Record<string, any>  // 动作参数
    }[]
  }[]

  // 动画配置（可选，支持入场、出场、交互动画）
  animations?: {
    enter?: {
      type: 'fade' | 'slide' | 'scale' | 'bounce'  // 入场动画类型
      duration: number  // 持续时间（毫秒）
      delay?: number  // 延迟（毫秒）
      easing?: string  // 缓动函数
    }
    exit?: {
      type: 'fade' | 'slide' | 'scale' | 'bounce'
      duration: number
      delay?: number
      easing?: string
    }
    interactions?: {
      event: string  // 触发事件，如 'hover', 'click'
      animation: {
        type: string
        duration: number
        properties: Record<string, any>  // CSS 属性变化
      }
    }[]
  }

  // 事件处理器（可选，定义组件事件响应）
  events?: {
    event: string  // 事件名，如 'click', 'dblclick', 'mouseover'
    actions: {
      type: 'script' | 'api' | 'linkage' | 'animation'  // 动作类型
      script?: string  // 自定义脚本
      api?: {
        url: string
        method: string
        params?: Record<string, any>
      }
      linkage?: {
        targetId: string
        action: string
        params?: Record<string, any>
      }
      animation?: {
        type: string
        duration: number
        properties: Record<string, any>
      }
    }[]
  }[]

  // 样式扩展（可选，覆盖默认样式）
  styles?: Record<string, any>

  // 权限控制（可选）
  permissions?: {
    visible?: string[]  // 可见角色
    editable?: string[]  // 可编辑角色
  }

  // 元数据（可选，用于版本控制、审计）
  metadata?: {
    createdAt: string
    updatedAt: string
    author: string
    version: string
  }
}

```
